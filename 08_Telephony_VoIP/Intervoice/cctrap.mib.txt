         CCTRAPS-MIB DEFINITIONS ::= BEGIN

                 MODULE-IDENTITY, NOTIFICATION-TYPE, OBJECT-TYPE, enterprises

         cctraps OBJECT IDENTIFIER ::= { intervoice  1 }

         ccResendTrapsFrom OBJECT-TYPE
             SYNTAX INTEGER
             MAX-ACCESS read-write
             STATUS current
             DESCRIPTION
                    "Alert will resend all traps after this sequence number."
             ::= { cctraps 3 }

                          cleared(1),

             MAX-ACCESS  read-only
             STATUS  current
             DESCRIPTION

         ccTrapResourceID OBJECT-TYPE

             MAX-ACCESS  read-only

             DESCRIPTION
                    "resource ID."

         ccTrapProbableCause OBJECT-TYPE
             SYNTAX  DisplayString
             MAX-ACCESS  read-only
             STATUS  current
             DESCRIPTION

         ccTrapSuggestedCorrectiveAction OBJECT-TYPE
             SYNTAX  DisplayString
             MAX-ACCESS  read-only

             DESCRIPTION
