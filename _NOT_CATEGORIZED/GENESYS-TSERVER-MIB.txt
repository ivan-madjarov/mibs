---     'gpPolling' object
---
        gpPollingStatus OBJECT-TYPE
                SYNTAX  DisplayString (SIZE(1..10))
                ACCESS  read-write
                STATUS  mandatory
                DESCRIPTION 
                "Activates and deactivates server polling feature, which
                sends the NMS periodical signals describing server's 
                operational status. Possible settings on/off."
                ::= { gpPolling 1 }

        gpPollingInterval       OBJECT-TYPE
                SYNTAX  INTEGER (0..120)
                ACCESS  read-write
                STATUS  mandatory
                DESCRIPTION 
                "Specifies the interval in seconds between polling signals from server."

                ::= { gpPolling 2 }

        gpPollingID     OBJECT-TYPE
                SYNTAX  INTEGER
                ACCESS  read-write
                STATUS  mandatory
                DESCRIPTION 
                "The event ID assigned to each polling signal, for monitoring purposes."
                ::= { gpPolling 3 }

        gpPollingLastTrap       OBJECT-TYPE
                SYNTAX  DisplayString (SIZE(1..64))
                ACCESS  read-only
                STATUS  mandatory
                DESCRIPTION 
                "The last ID assigned to the polling signal."
                ::= { gpPolling 4 }

        gpPollingSignal TRAP-TYPE
                ENTERPRISE      server
                VARIABLES       { gpPollingLastTrap }
                DESCRIPTION
                        "SNMP polling signal received from server."
                ::= 2
